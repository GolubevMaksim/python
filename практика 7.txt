def palindrome(s):
    if len(s) < 1:
        return True
    else:
        if s[0] == s[-1]:
            return palindrome(s[1:-1])
        else:
            return False

def power(a, n):
    if (n == 0):
        return (1)
    if n % 2 == 0:
        return power(a, n/2)**2
    else:
        return (a*power(a, n-1))

def fib(n):
    if n in (1, 2):
        return 1
        return fib(n - 1) + fib(n - 2)
